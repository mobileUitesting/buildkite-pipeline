import android.view.View;
import androidx.test.espresso.Espresso;
import androidx.test.espresso.action.ViewActions;
import androidx.test.espresso.matcher.ViewMatchers;
import androidx.test.ext.junit.runners.AndroidJUnit4;
import androidx.test.filters.LargeTest;
import androidx.test.rule.ActivityTestRule;
import org.junit.Rule;
import org.junit.Test;
import org.junit.runner.RunWith;

import static androidx.test.espresso.assertion.ViewAssertions.matches;
import static androidx.test.espresso.matcher.ViewMatchers.isDisplayed;
import static androidx.test.espresso.matcher.ViewMatchers.withId;
import static androidx.test.espresso.matcher.ViewMatchers.withText;

@RunWith(AndroidJUnit4.class)
@LargeTest
public class ExampleEspressoTest {

    @Rule
    public ActivityTestRule<MainActivity> activityRule = new ActivityTestRule<>(MainActivity.class);

    @Test
    public void testButtonClick() {
        // Find the EditText by its ID and type a text
        Espresso.onView(withId(R.id.editText))
                .perform(ViewActions.typeText("Hello, Espresso!"));

        // Close the soft keyboard
        Espresso.onView(withId(R.id.editText))
                .perform(ViewActions.closeSoftKeyboard());

        // Find the Button by its ID and click it
        Espresso.onView(withId(R.id.buttonSubmit))
                .perform(ViewActions.click());

        // Verify that the result TextView is displayed and has the correct text
        Espresso.onView(withId(R.id.textViewResult))
                .check(matches(isDisplayed()))
                .check(matches(withText("Hello, Espresso!")));
    }
}
